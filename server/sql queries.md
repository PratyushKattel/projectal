CREATE DATABASE projectaldb;

CREATE TABLE workspace(
ws_id INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
name VARCHAR(50) NOT NULL,
owner_id INT NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
FOREIGN KEY(owner_id) REFERENCES auth_user(id)
);

CREATE TYPE role_name AS ENUM ('Owner', 'Admin', 'Member', 'Viewer');

CREATE TABLE role (
role_id SERIAL PRIMARY KEY,
name role_name NOT NULL DEFAULT 'Member',
ws_id INTEGER REFERENCES workspace(ws_id) ON DELETE CASCADE,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE ws_member (
member_id SERIAL PRIMARY KEY,
user_id INTEGER REFERENCES auth_user(id) ON DELETE CASCADE,
ws_id INTEGER REFERENCES workspace(ws_id) ON DELETE CASCADE,
role_id INTEGER REFERENCES role(role_id) ON DELETE SET NULL,
joined_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE workspace_invite (
invite_id SERIAL PRIMARY KEY,
email VARCHAR(255) NOT NULL,
ws_id INTEGER REFERENCES workspace(ws_id) ON DELETE CASCADE,
token UUID NOT NULL UNIQUE,
invited_by INTEGER REFERENCES auth_user(id),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
expires_at TIMESTAMP
);
